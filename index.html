// --- Configuration & State --- const MIN_W = 72; const COLORS = [ { name: 'rose', bg: '#f43f5e', glow: '#f43f5e55' }, { name: 'amber', bg: '#f59e0b', glow: '#f59e0b55' }, { name: 'emerald', bg: '#10b981', glow: '#10b98155' }, { name: 'blue', bg: '#3b82f6', glow: '#3b82f655' }, { name: 'purple', bg: '#a855f7', glow: '#a855f755' }, { name: 'pink', bg: '#ec4899', glow: '#ec489955' }, { name: 'teal', bg: '#14b8a6', glow: '#14b8a655' }, { name: 'orange', bg: '#f97316', glow: '#f9731655' }, { name: 'indigo', bg: '#6366f1', glow: '#6366f155' }, { name: 'lime', bg: '#84cc16', glow: '#84cc1655' } ]; let state = { widths: Array(24).fill(112), tasks: [], isEditing: false, editName: '', editNote: '', activeColorIdx: 0, draggingId: null, dropHour: null, activeModalTask: null, resizing: false }; // --- Helpers --- const $ = (id) => document.getElementById(id); const getHourLabel = (h) => { const m = h < 12 ? 'AM' : 'PM'; const displayH = h % 12 === 0 ? 12 : h % 12; return `${displayH} ${m}`; }; const getHourStyle = (h) => { if (h >= 9 && h <= 16) return '#a78bfa'; if (h >= 6 && h <= 18) return '#4a5f8a'; return '#2a3a5c'; }; const getPeriodEmoji = (h) => { if (h <= 5) return '🌙'; if (h <= 11) return '🌤'; if (h <= 17) return '☀️'; return '🌆'; }; const getColumnBg = (h) => { if (h <= 5) return 'rgba(10,5,30,0.6)'; if (h <= 11) return 'rgba(15,10,5,0.4)'; if (h <= 17) return 'rgba(10,5,25,0.2)'; return 'rgba(5,10,25,0.5)'; }; // --- Core Functions --- function render() { const grid = $('grid-container'); const dock = $('dock-items'); // Stats const placedCount = state.tasks.filter(t => t.hour !== null).length; $('stat-pill').innerText = `${placedCount} / ${state.tasks.length} placed`; // Grid Rendering grid.innerHTML = ''; state.widths.forEach((w, i) => { const col = document.createElement('div'); col.className = 'hour-col'; col.style.width = `${w}px`; col.style.backgroundColor = getColumnBg(i); const header = document.createElement('div'); header.className = 'col-header'; header.innerHTML = ` ${getPeriodEmoji(i)} ${getHourLabel(i)} `; const dropZone = document.createElement('div'); dropZone.className = 'drop-zone'; if (state.dropHour === i) dropZone.classList.add('drag-over'); // Add existing tasks to this hour state.tasks.filter(t => t.hour === i).forEach(task => { const bubble = document.createElement('div'); bubble.className = 'task-bubble'; bubble.style.backgroundColor = task.color.bg; bubble.innerText = task.name; bubble.onclick = () => { state.activeModalTask = task; render(); }; dropZone.appendChild(bubble); }); // Show drop indicator ONLY if we are dragging something and hovering this column if (state.dropHour === i && state.draggingId) { const indicator = document.createElement('div'); indicator.className = 'drop-indicator'; indicator.innerText = 'drop here'; dropZone.appendChild(indicator); } // Drag Events dropZone.addEventListener('dragover', (e) => { e.preventDefault(); // Required to allow drop if (state.dropHour !== i) { state.dropHour = i; // Use requestAnimationFrame or a slight delay if flickering occurs // during state updates, but here we just re-render. render(); } }); dropZone.addEventListener('dragleave', (e) => { // Check if we are really leaving the zone (not just moving over a child) if (e.relatedTarget && !dropZone.contains(e.relatedTarget)) { state.dropHour = null; render(); } }); dropZone.addEventListener('drop', (e) => { e.preventDefault(); const task = state.tasks.find(t => t.id === state.draggingId); if (task) { task.hour = i; } state.draggingId = null; state.dropHour = null; render(); }); col.appendChild(header); col.appendChild(dropZone); if (i < 23) { const handle = document.createElement('div'); handle.className = 'resize-handle'; handle.innerHTML = '
'; handle.onmousedown = (e) => initResize(e, i); col.appendChild(handle); } grid.appendChild(col); }); // Dock Rendering dock.innerHTML = ''; state.tasks.filter(t => t.hour === null).forEach(task => { const bubble = document.createElement('div'); bubble.className = 'dock-bubble'; bubble.draggable = true; bubble.innerText = task.name; bubble.style.backgroundColor = task.color.bg; bubble.style.boxShadow = `0 4px 12px ${task.color.glow}`; bubble.addEventListener('dragstart', (e) => { state.draggingId = task.id; // Native ghost image sometimes fails if the element is modified immediately. // We use a timeout to change opacity so the ghost is captured first. setTimeout(() => { bubble.style.opacity = '0.35'; }, 0); e.dataTransfer.effectAllowed = 'move'; }); bubble.addEventListener('dragend', () => { bubble.style.opacity = '1'; state.draggingId = null; state.dropHour = null; render(); }); dock.appendChild(bubble); }); // "New Task" Input Mode renderInputMode(dock); // Modal renderModal(); } function renderInputMode(container) { if (state.isEditing) { const inputPill = document.createElement('div'); inputPill.className = 'input-pill'; inputPill.innerHTML = `
${COLORS.map((c, idx) => `
`).join('')}
`; container.appendChild(inputPill); const nameIn = $('task-name-in'); nameIn.focus(); nameIn.addEventListener('input', (e) => state.editName = e.target.value); $('task-note-in').addEventListener('input', (e) => state.editNote = e.target.value); inputPill.onkeydown = (e) => { if(e.key === 'Enter') saveTask(); if(e.key === 'Escape') { state.isEditing = false; render(); } }; } else { const addBtn = document.createElement('button'); addBtn.className = 'btn-new-task'; addBtn.innerHTML = `＋ New task`; addBtn.onclick = () => { state.isEditing = true; render(); }; container.appendChild(addBtn); } } function renderModal() { const modalCont = $('modal-container'); modalCont.innerHTML = ''; if (state.activeModalTask) { const task = state.activeModalTask; const overlay = document.createElement('div'); overlay.className = 'modal-overlay'; overlay.onclick = () => { state.activeModalTask = null; render(); }; overlay.innerHTML = `
📌

${task.name}

Scheduled · ${getHourLabel(task.hour)}

${task.note ? `
${task.note}
` : ''}
`; modalCont.appendChild(overlay); } } function saveTask() { if (!state.editName.trim()) return; state.tasks.push({ id: 't-' + Date.now(), name: state.editName, note: state.editNote, color: COLORS[state.activeColorIdx], hour: null }); state.editName = ''; state.editNote = ''; state.isEditing = false; state.activeColorIdx = (state.activeColorIdx + 1) % COLORS.length; render(); } function returnToDock(id) { const task = state.tasks.find(t => t.id === id); if (task) task.hour = null; state.activeModalTask = null; render(); } function deleteTask(id) { state.tasks = state.tasks.filter(t => t.id !== id); state.activeModalTask = null; render(); } function initResize(e, index) { state.resizing = true; document.body.style.cursor = 'col-resize'; const startX = e.clientX; const leftW0 = state.widths[index]; const rightW0 = state.widths[index + 1]; const total = leftW0 + rightW0; function onMouseMove(mE) { const dx = mE.clientX - startX; const newLeft = Math.max(MIN_W, Math.min(total - MIN_W, leftW0 + dx)); state.widths[index] = newLeft; state.widths[index + 1] = total - newLeft; render(); } function onMouseUp() { state.resizing = false; document.body.style.cursor = 'default'; document.removeEventListener('mousemove', onMouseMove); document.removeEventListener('mouseup', onMouseUp); } document.addEventListener('mousemove', onMouseMove); document.addEventListener('mouseup', onMouseUp); } // Start render();
